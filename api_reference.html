Julia API Reference ​

This document describes the programmatic interface for the Julia Auction System. The system is designed as a Julia library for auction research and simulation.

Getting Started ​

julia
using Pkg
Pkg.activate(".")  # Activate the project environment

# Load the main modules
include("src/AuctionSimulator.jl")
include("src/AugmentedUniformPriceAuction.jl")

# Or use the simulation interface
include("src/simulation.jl")

Core Modules ​

AuctionSimulator.jl ​

The main simulation framework for running auctions with different bidding strategies.

Key Types ​

julia
# Abstract base types
abstract type AbstractAuction end
abstract type AbstractBidder end 
abstract type AbstractBiddingStrategy end

# Core data structures
struct Bid
    bidder_id::Int
    value::Float64
    timestamp::DateTime
end

struct AuctionResult{T <: AbstractAuction}
    auction_type::T
    winner_id::Union{Int, Nothing}
    winning_price::Float64
    all_bids::Vector{Bid}
    statistics::StatisticsDict
end

# Bidder implementation
struct Bidder{S <: AbstractBiddingStrategy} <: AbstractBidder
    id::Int
    valuation::Float64
    strategy::S
end

Auction Types ​

julia
# First-price sealed-bid auction
struct FirstPriceAuction <: AbstractAuction
    reserve_price::Float64
    items::Int
end

# Second-price (Vickrey) auction  
struct SecondPriceAuction <: AbstractAuction
    reserve_price::Float64
    items::Int
end

Bidding Strategies ​

julia
# Truthful bidding (bid true valuation)
struct TruthfulBidder <: AbstractBiddingStrategy end

# Bid shading strategy
struct ShadedBidder <: AbstractBiddingStrategy
    shading_factor::Float64  # Bid = valuation * (1 - shading_factor)
end

# Random bidding
struct RandomBidder <: AbstractBiddingStrategy
    min_factor::Float64
    max_factor::Float64
end

# Strategic bidding based on auction type
struct StrategicBidder <: AbstractBiddingStrategy end

Usage Examples ​

Basic First-Price Auction ​

julia
# Create bidders with different strategies
bidders = [
    Bidder(1, 50.0, TruthfulBidder()),
    Bidder(2, 45.0, ShadedBidder(0.1)),
    Bidder(3, 60.0, StrategicBidder())
]

# Create auction
auction = FirstPriceAuction(10.0, 1)

# Run auction
result = conduct_auction(bidders, auction)

# Analyze results
println("Winner: $(result.winner_id)")
println("Price: $(result.winning_price)")

Augmented Uniform Price Auction ​

julia
# Create supply schedule
points = [(10.0, 100.0), (50.0, 500.0), (100.0, 800.0)]
schedule = create_elastic_schedule(points, EXPONENTIAL)

# Configure auction
config = AuctionConfig(
    reserve_price=10.0,
    supply_schedule=schedule,
    allow_partial_fills=true,
    tie_breaking=AugmentedTieBreaking()
)

# Create bids
bids = [
    Bid(1, 45.0, now()),
    Bid(2, 50.0, now()),
    Bid(3, 35.0, now())
]

# Run auction
result = run_auction(bids, config)

Error Handling ​

The system defines custom error types for different failure modes:

julia
# Settlement system errors
abstract type SettlementError <: Exception end

struct ValidationError <: SettlementError
    message::String
    batch_id::UUID
    details::Dict{String, Any}
end

struct ExecutionError <: SettlementError
    message::String
    batch_id::UUID
    step::Int
    cause::Union{Exception, Nothing}
end

# Configuration errors
struct ConfigError <: Exception
    message::String
    key::Union{String, Nothing}
    cause::Union{Exception, Nothing}
end

Performance Considerations ​

Type Stability ​

The system is designed for type stability with minimal use of Any types:

julia
# Good - type stable
function calculate_revenue(bids::Vector{Bid}) -> Float64
    return sum(bid.value for bid in bids)
end

# Avoid - type unstable  
function get_winner(result::AuctionResult) -> Union{Int, Nothing}
    return result.winner_id  # May be Nothing, but semantically correct
end

Memory Usage ​

  • Bid vectors are preallocated when possible

  • Large simulations use streaming processing

  • Results can be serialized for analysis

Parallelization ​

julia
# Enable parallel processing
using Distributed
addprocs(4)

# Run parallel simulation
result = run_simulation(auction, bidders, SimulationParams(parallel=true))
